    ```caddyfile
    #
    # Host topology:
    #   host nginx (:80/:443, TLS)
    #     â””â†’ 127.0.0.1:${CADDY_INTERNAL_PORT:-8080} (this Caddy)
    #         â”śâ†’ app:3000            (Next.js dashboard + API)
    #         â”śâ†’ terminal:3001       (Socket.IO â€” /ws/*)
    #         â””â†’ devdock-{slug}-dev  (per-project preview â€” registered at runtime via Admin API)
    #
    # Security:
    #   - D-03: HTTP-only inside the stack. Host nginx handles TLS. No cert volumes here.
    #   - Admin API on :2019 binds to 0.0.0.0 for compose-internal reachability,
    #     BUT port 2019 is NEVER published to the host (T-999.2-02).
    #   - RESEARCH.md Pitfall 3: default `localhost:2019` is container-local â€” unreachable
    #     from `app`. `admin :2019` fixes this by binding to all interfaces.

    {
        # Admin API â€” compose-internal only (T-999.2-02).
        admin :2019

        # No automatic HTTPS issuance â€” host nginx terminates TLS (D-03).
        auto_https off

        # Persist dynamic config changes (preview routes added via Admin API) so they
        # survive container restarts. Stored in the caddy_data volume mounted in compose.
        persist_config on

        # JSON logging for journalctl-friendly output.
        log {
            output stdout
            format json
        }
    }

    # Public entry point â€” receives traffic from host nginx.
    :8080 {
        # Socket.IO path (terminal auth + /logs namespace â€” see server/terminal-server.ts).
        # Both the /terminal and /logs namespaces share the `/ws/socket.io` path.
        handle /ws/* {
            reverse_proxy terminal:3001
        }

        # Everything else â†’ Next.js app (dashboard, API, preview subdomain rewrites
        # handled internally by src/middleware.ts for non-Caddy-registered hosts).
        handle {
            reverse_proxy app:3000
        }
    }
    ```

    Notes:

    - `auto_https off` + only `:8080` (no `:443`) means Caddy never attempts to acquire certs.
