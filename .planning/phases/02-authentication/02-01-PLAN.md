**Step 5 -- Create `src/auth.ts`** (Full Node.js config with Credentials + bcrypt + DB):
```typescript
import NextAuth from 'next-auth';
import Credentials from 'next-auth/providers/credentials';
import bcrypt from 'bcrypt';
import { db } from '@/lib/db';
import { users } from '@/lib/db/schema';
import { eq } from 'drizzle-orm';
import { authConfig } from './auth.config';
import { checkRateLimit, recordFailedAttempt, clearFailedAttempts } from '@/lib/auth/rate-limit';

export const { handlers, signIn, signOut, auth } = NextAuth({
  ...authConfig,
  providers: [
    Credentials({
      credentials: {
        username: { label: 'Username', type: 'text' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        if (!credentials?.username || !credentials?.password) return null;

        const username = credentials.username as string;
        const password = credentials.password as string;

        // Rate limit check per D-13
        const rateCheck = checkRateLimit(username);
        if (!rateCheck.allowed) {
          throw new Error('RATE_LIMITED');
        }

        const [user] = await db
          .select()
          .from(users)
          .where(eq(users.username, username))
          .limit(1);

        if (!user) {
          recordFailedAttempt(username);
          return null;
        }

        const passwordValid = await bcrypt.compare(password, user.passwordHash);

        if (!passwordValid) {
          recordFailedAttempt(username);
          return null;
        }

        // Successful login -- clear failed attempts
        clearFailedAttempts(username);

        return {
          id: user.id,
          name: user.username,
          role: user.role,
        };
      },
    }),
  ],
  callbacks: {
    ...authConfig.callbacks,
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id!;
        token.role = (user as { role?: string }).role ?? 'user';
      }
      return token;
    },
    async session({ session, token }) {
      if (token) {
        session.user.id = token.id;
        (session.user as { role: string }).role = token.role;
      }
      return session;
    },
  },
});
```

**Step 1 -- Create `src/lib/auth/rate-limit.ts`** (per D-13: 5 attempts, 30-second cooldown):
```typescript
// In-memory rate limiter for login attempts
// Resets on server restart (acceptable per D-13)

const failedAttempts = new Map<string, { count: number; lastAttempt: number }>();

const MAX_ATTEMPTS = 5;
const COOLDOWN_MS = 30 * 1000; // 30 seconds

export function checkRateLimit(username: string): { allowed: boolean; retryAfter?: number } {
  const record = failedAttempts.get(username);
  if (!record) return { allowed: true };

  const elapsed = Date.now() - record.lastAttempt;
  if (record.count >= MAX_ATTEMPTS && elapsed < COOLDOWN_MS) {
    return { allowed: false, retryAfter: Math.ceil((COOLDOWN_MS - elapsed) / 1000) };
  }

  // Cooldown expired -- reset
  if (elapsed >= COOLDOWN_MS) {
    failedAttempts.delete(username);
    return { allowed: true };
  }

  return { allowed: true };
}

export function recordFailedAttempt(username: string): void {
  const record = failedAttempts.get(username) || { count: 0, lastAttempt: 0 };
  record.count += 1;
  record.lastAttempt = Date.now();
  failedAttempts.set(username, record);
}

export function clearFailedAttempts(username: string): void {
  failedAttempts.delete(username);
}
```

**Step 2 -- Create `src/lib/auth/actions.ts`** (server actions for form integration):

```typescript
'use server';

import { signIn, signOut } from '@/auth';
import { AuthError } from 'next-auth';

export async function login(
  prevState: { error: string | null },
  formData: FormData
): Promise<{ error: string | null }> {
  try {
    await signIn('credentials', {
      username: formData.get('username'),
      password: formData.get('password'),
      redirectTo: '/dashboard',
    });
    return { error: null };
  } catch (error) {
    if (error instanceof AuthError) {
      if (error.type === 'CredentialsSignin') {
        return { error: 'Invalid username or password.' }; // D-04: generic message
      }
      // Check for rate limit error from authorize function
      if (error.message?.includes('RATE_LIMITED') || error.cause?.err?.message?.includes('RATE_LIMITED')) {
        return { error: 'Too many attempts. Try again in 30 seconds.' };
      }
      return { error: 'Something went wrong. Please try again.' };
    }
    throw error; // Re-throw redirect errors (signIn throws NEXT_REDIRECT on success)
  }
}

export async function logout(): Promise<void> {
  await signOut({ redirectTo: '/login' }); // D-17
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> Middleware | All HTTP requests pass through Next.js middleware for auth check |
| Login Form -> authorize() | Username/password crosses from untrusted client to server-side credential verification |
| JWT Cookie -> Server | Session token in HTTP-only cookie is presented on every request |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | authorize() | mitigate | bcrypt.compare for password verification (12 salt rounds); rate limiting at 5 attempts/30s per username |
| T-02-02 | Tampering | /api/auth/* | mitigate | Auth.js built-in double-submit cookie CSRF protection; session strategy JWT with AUTH_SECRET signing |
| T-02-03 | Repudiation | login action | accept | No audit log in this phase; single-user tool, acceptable risk |
| T-02-04 | Information Disclosure | error messages | mitigate | Generic "Invalid username or password." message per D-04; never reveal which field is wrong; no stack traces in responses |
| T-02-05 | Denial of Service | rate-limit.ts | mitigate | In-memory rate limiter blocks after 5 failures for 30 seconds per D-13; Map cleanup on cooldown expiry |
| T-02-06 | Elevation of Privilege | middleware.ts | mitigate | All /dashboard/* and /api/* routes require valid JWT session; /api/health and /api/auth/* explicitly excluded; session regeneration on login (Auth.js default) |
