    /**
     * Generate a URL-safe slug from an environment name.
     * Lowercase, replace non-alphanumeric with hyphens, collapse multiple hyphens, trim hyphens.
     */
    export function generateSlug(name: string): string {
      return name
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, '-')
        .replace(/^-+|-+$/g, '')
        .slice(0, 63); // Docker project names have practical limits
    }

    /**
     * Validate that a slug contains only safe characters.
     * Used to verify slugs before passing to Docker CLI or filesystem operations.
     */
    export function isValidSlug(slug: string): boolean {
      return /^[a-z0-9]([a-z0-9-]*[a-z0-9])?$/.test(slug) && slug.length > 0 && slug.length <= 63;
    }
    ```

       const [newEnv] = await db.insert(environments).values({
         userId: session.user.id,
         name: parsed.data.name,
         slug,
         repoUrl: parsed.data.repoUrl || null,
         status: 'starting',
         composeConfig: { enablePostgres: parsed.data.enablePostgres, enableRedis: parsed.data.enableRedis },
         dockerProjectName,
         networkName,
       }).returning();
       ```

    IMPORTANT: The background operation MUST NOT block the HTTP response. The 202 response returns immediately. The polling UI (D-05) picks up status changes.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> API routes | Untrusted HTTP requests cross auth boundary |
| API routes -> Docker service | Validated data passes to Docker CLI execution |
| API routes -> Database | User-scoped queries prevent IDOR |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05 | Spoofing | All routes | mitigate | Every route calls `await auth()` and checks `session.user.id`. Returns 401 if no session. |
| T-03-06 | Spoofing (IDOR) | All routes | mitigate | All DB queries include `eq(environments.userId, session.user.id)`. Users cannot access, modify, or delete other users' environments. |
| T-03-07 | Tampering | POST /api/environments | mitigate | Input validated via zod schema. Slug validated with `isValidSlug()` regex `[a-z0-9-]`. RepoUrl validated as URL by zod. |
| T-03-08 | Denial of Service | POST /api/environments | mitigate | `DEVDOCK_MAX_CONCURRENT_ENVS` check before creation. Same check in start route. Prevents resource exhaustion. |
| T-03-09 | Tampering | DELETE route | mitigate | Status guard prevents deletion during transitions (starting/stopping). Prevents orphaned Docker resources from race conditions. |
| T-03-10 | Information Disclosure | Error responses | mitigate | Docker stderr errors truncated to 500 chars in DB. Client responses use generic messages from copywriting contract, not raw Docker output. |
</threat_model>
